A map
Add in the header: 
<script type="text/javascript" src="${request.route_url('apiloader')}"></script>
<script>
    window.onload = function() {
        var map1 = new c2cgeoportal.Map({
            div: 'map1',
            zoom: 12,
            easting: 645513,
            northing: 259897,
            height: 400,
            width: 600
        });
    };
</script>
Add in the body where you want the map: 
<div id='map1'></div>

A map with a subset of overlays
Add in the header: 
<script type="text/javascript" src="${request.route_url('apiloader')}"></script>
<script>
    window.onload = function() {
        var map3 = new c2cgeoportal.Map({
            div: 'map3',
            zoom: 14,
            easting: 663382,
            northing: 244914,
            height: 500,
            width: 800,
            overlays: {
                'local': [
                    'TOPIC_Liegenschaften_ALL', // example of theme
                    'strom_station_grundriss' // example of layer
                ],
                'external': [
                    'TOPIC_Liegenschaften_LN'
                ]
            }
        });
    };
</script>
Add in the body where you want the map: 
<div id='map3'></div>
